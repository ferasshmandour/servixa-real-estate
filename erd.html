    users {
        bigint id PK
        string first_name
        string last_name
        string phone UK
        string email UK
        string country
        string password
        boolean is_verified
        string profile_image
        string device_token
    }

    admins {
        bigint id PK
        string name
        string email UK
        string password
        bigint role_id FK
    }

    cities {
        bigint id PK
        string name_ar
        string name_en
    }

    activity_types {
        bigint id PK
        string name_ar
        string name_en
    }

    business_accounts {
        bigint id PK
        bigint user_id FK
        bigint activity_type_id FK
        bigint city_id FK
        string license_number
        string name_ar
        string name_en
        text activities
        text details
        text address
        decimal latitude
        decimal longitude
        enum status "pending/approved/rejected"
        text rejection_reason
    }

    business_account_files {
        bigint id PK
        bigint business_account_id FK
        string file_path
        enum file_type "document/image"
    }

    categories {
        bigint id PK
        bigint parent_id FK "self-ref nullable"
        string name_ar
        string name_en
        string icon
        integer sort_order
    }

    dynamic_fields {
        bigint id PK
        bigint category_id FK
        string label_ar
        string label_en
        enum field_type "text/number/select/textarea/boolean"
        json options
        boolean is_required
        integer sort_order
    }

    sliders {
        bigint id PK
        string image
        string link
        integer sort_order
        boolean is_active
    }

    services {
        bigint id PK
        bigint business_account_id FK
        bigint category_id FK
        bigint subcategory_id FK "nullable"
        string title_ar
        string title_en
        text description_ar
        text description_en
        integer available_quantity
        string main_image
        enum type "sale/rent"
        decimal price
        enum currency "USD/SYP"
        decimal latitude
        decimal longitude
        enum status "pending/approved/rejected"
        text rejection_reason
    }

    service_images {
        bigint id PK
        bigint service_id FK
        string image_path
    }

    service_dynamic_values {
        bigint id PK
        bigint service_id FK
        bigint dynamic_field_id FK
    }

    orders {
        bigint id PK
        bigint service_id FK
        bigint requester_business_id FK
        date needed_at
        integer quantity
        text details
        enum status "pending/accepted/rejected"
    }

    ratings {
        bigint id PK
        bigint order_id FK "UNIQUE"
        bigint service_id FK
        bigint user_id FK
        tinyint rating
        text comment
    }

    favorites {
        bigint id PK
        bigint user_id FK
        bigint service_id FK
    }

    reports {
        bigint id PK
        bigint user_id FK
        bigint service_id FK
        text reason
        boolean is_resolved
    }

    conversations {
        bigint id PK
        bigint service_id FK
        bigint initiator_id FK
        bigint receiver_id FK
    }

    messages {
        bigint id PK
        bigint conversation_id FK
        bigint sender_id FK
        text content
        enum status "sent/read"
    }

    notifications {
        bigint id PK
        bigint user_id FK
        string title
        text body
        json data
        timestamp read_at
    }

    %% ===== RELATIONSHIPS =====

    %% Users core
    users ||--o{ business_accounts : "owns"
    users ||--o{ favorites : "has"
    users ||--o{ ratings : "writes"
    users ||--o{ reports : "submits"
    users ||--o{ notifications : "receives"
    users ||--o{ conversations : "initiates"
    users ||--o{ conversations : "receives"
    users ||--o{ messages : "sends"

    %% Admins
    admins }o--|| roles : "has role"

    %% Business Accounts
    business_accounts }o--|| activity_types : "has type"
    business_accounts }o--|| cities : "located in"
    business_accounts ||--o{ business_account_files : "has files"
    business_accounts ||--o{ services : "posts"
    business_accounts ||--o{ orders : "requests"

    %% Categories (self-referential)
    categories ||--o{ categories : "parent has subcategories"
    categories ||--o{ dynamic_fields : "has fields"
    categories ||--o{ services : "categorizes"

    %% Services
    services ||--o{ service_images : "has images"
    services ||--o{ service_dynamic_values : "has values"
    services ||--o{ orders : "receives"
    services ||--o{ ratings : "rated"
    services ||--o{ favorites : "favorited"
    services ||--o{ reports : "reported"
    services ||--o{ conversations : "discussed in"

    %% Dynamic Fields
    dynamic_fields ||--o{ service_dynamic_values : "filled by"

    %% Orders
    orders ||--o| ratings : "has one rating"

    %% Conversations
    conversations ||--o{ messages : "contains"
